### `src/i18n/index.ts`

```typescript
import en from "./en";
import ja from "./ja";
import type { Translations } from "./en";

export type Locale = "en" | "ja";
export type { Translations };

export const translations: Record<Locale, Translations> = { en, ja };

export function detectLocale(): Locale {
  const lang = navigator.language?.slice(0, 2);
  return lang === "ja" ? "ja" : "en";
}
```

### `src/store/i18nSlice.ts`

Redux slice for the current locale:

```typescript
import { createSlice, PayloadAction } from "@reduxjs/toolkit";
import type { Locale } from "../i18n";
import { detectLocale } from "../i18n";

const LOCALE_KEY = "philharmonic.webui.locale";

function storedLocale(): Locale {
  try {
    const stored = sessionStorage.getItem(LOCALE_KEY);
    if (stored === "en" || stored === "ja") return stored;
  } catch {}
  return detectLocale();
}

interface I18nState { locale: Locale; }

const i18nSlice = createSlice({
  name: "i18n",
  initialState: { locale: storedLocale() } as I18nState,
  reducers: {
    setLocale(state, action: PayloadAction<Locale>) {
      state.locale = action.payload;
      try { sessionStorage.setItem(LOCALE_KEY, action.payload); } catch {}
    },
  },
});

export const { setLocale } = i18nSlice.actions;
export default i18nSlice.reducer;
```

### `src/hooks/useT.ts`

```typescript
import { useAppSelector } from "../store";
import { translations } from "../i18n";
import type { Translations } from "../i18n";

export function useT(): Translations {
  const locale = useAppSelector((state) => state.i18n.locale);
  return translations[locale];
}
```

Every user-visible string must be in the translation files.
Do NOT leave any hardcoded English in the TSX files except:
- Technical identifiers (UUID, pht_, HTTP methods)
- CSS class names
- JSON keys

- Japanese translations should be natural, not machine-literal.
  Use polite-neutral register (です/ます form where applicable,
  but brief labels use noun form).
- Navigation labels should be concise (1-2 words).
- Error messages should be clear and actionable.
